   - `PLACES_FIELD_MASK` = exactly: `"places.id,places.displayName,places.location,places.types,places.primaryType,places.primaryTypeDisplayName,places.businessStatus,places.formattedAddress,places.shortFormattedAddress,places.addressComponents"` (NO rating, NO userRatingCount)
